import React, { useState, useEffect } from 'react';
import './App.css';

function App() {
  // State: This is the app's memory
  const [weather, setWeather] = useState(null);
  const [location, setLocation] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const API_KEY = 'YH2CSKLQJTQMW7M8HQ43GNJ7HG';

  const fetchWeather = async (city) => {
    if (!city) return;
    
    setLoading(true);
    setError(null);
    
    try {
      // Requirements call for current, previous 24h, and future 24h
      const response = await fetch(
        `https://weather.visualcrossing.com/VisualCrossingWebServices/rest/services/timeline/${city}?unitGroup=metric&key=${API_KEY}&contentType=json`
      );
      
      if (!response.ok) throw new Error('Location not found');
      
      const data = await response.json();
      setWeather(data); 
      console.log(data); // Look at your console to see the "treasure chest" of data
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  // Stretch Goal: Default to current location on load
  useEffect(() => {
    // We will add the Geolocation logic here next!
  }, []);

  return (
    <div className="app-container">
      <h1>Sky Scout</h1>
      
      <div className="search-box">
        <input 
          type="text" 
          placeholder="Enter location..." 
          value={location}
          onChange={(e) => setLocation(e.target.value)}
        />
        <button onClick={() => fetchWeather(location)}>Search</button>
        {/* Requirement: User can refresh the outlook */}
        <button onClick={() => fetchWeather(location)}>Refresh</button>
      </div>

      {loading && <p>Loading weather data...</p>}
      {error && <p style={{ color: 'red' }}>{error}</p>}

      {weather && (
        <div className="weather-display">
          {/* We will build the CurrentWeather and Forecast components here */}
          <h2>{weather.resolvedAddress}</h2>
          <p>Current Temp: {weather.currentConditions.temp}Â°C</p>
        </div>
      )}
    </div>
  );
}

export default App;
